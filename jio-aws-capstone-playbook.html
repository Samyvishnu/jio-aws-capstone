TPM NOTE — IAM Philosophy
  • Never use AdministratorAccess on service roles
  • One role per service (EKS cluster role ≠ node role ≠ app role)
  • Use OIDC for pod-level AWS permissions (no keys in pods)
  • Tag every role with Project=jio-capstone for cost attribution

JSLambda@Edge — multi-CDN routing + graceful degradation
// Lambda@Edge: Origin Request handler
// Routes video → high-perf CDN, fallback on 5xx
exports.handler = async (event) => {
  const request = event.Records[0].cf.request;
  const uri = request.uri;

  // Route scores/promos to cost-efficient origin
  if (uri.includes('/scores/') || uri.includes('/promos/')) {
    request.origin.s3.path = '/static-fallback';
    return request;
  }

  // Graceful degradation: if origin returns 5xx, serve cached static
  if (event.Records[0].cf.response?.status?.startsWith('5')) {
    return {
      status: '200',
      headers: { 'content-type': [{value: 'image/png'}] },
      body: '<!-- serve scoreboard static asset -->',
      bodyEncoding: 'text'
    };
  }
  return request;
};

# 06-ai/traffic_predictor.py
import numpy as np
from sklearn.ensemble import GradientBoostingRegressor
from sklearn.preprocessing import StandardScaler
import json, boto3, os

# Synthetic IPL match viewer data
np.random.seed(42)
n = 1000
X = np.column_stack([
    np.random.choice([1,2,3,4,5,6,7], n),    # day of week
    np.random.randint(0, 24, n),                  # match start hour
    np.random.choice([1,0], n, p=[0.6,0.4]),     # is_weekend
    np.random.choice([1,0], n, p=[0.3,0.7]),     # is_final
    np.random.normal(25, 5, n)                     # historical_avg_M_viewers
])
y = X[:,4] * (1.2 + 0.5*X[:,2] + 1.8*X[:,3]) + np.random.normal(0,2,n)

scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)
model = GradientBoostingRegressor(n_estimators=100, learning_rate=0.1)
model.fit(X_scaled, y)

# Predict: IPL Final, Sunday 7PM, 30M historical avg
test_event = np.array([[7, 19, 1, 1, 30]])
predicted_m = model.predict(scaler.transform(test_event))[0]
replicas_needed = max(3, int(predicted_m / 2.5))

print(f"Predicted viewers: {predicted_m:.1f}M")
print(f"Recommended replicas: {replicas_needed}")

# Write to SSM (triggers EKS pre-scale)
ssm = boto3.client('ssm', region_name='ap-south-1')
ssm.put_parameter(
    Name='/jio-capstone/predicted-replicas',
    Value=str(replicas_needed),
    Type='String',
    Overwrite=True
)
print("✅ SSM updated → EKS will pre-scale")

#!/bin/bash
source .env
echo "⚠️  TEARDOWN STARTING — $(date)"

# 1. EKS FIRST ($0.10/hr cluster charge)

eksctl delete cluster --name jio-streaming-cluster --region ap-south-1 || true

# 2. MediaLive channels (~$15/hr)
echo "[2/8] Stopping MediaLive channels..."
for ch in $(aws mediapackage list-channels --query 'Channels[*].Id' --output text 2>/dev/null); do
  aws mediapackage delete-channel --id "$ch" 2>/dev/null
done

# 3. SageMaker endpoint
echo "[3/8] Deleting SageMaker resources..."
aws sagemaker delete-endpoint --endpoint-name jio-traffic-predictor-endpoint 2>/dev/null
aws sagemaker delete-endpoint-config --endpoint-config-name jio-predictor-config 2>/dev/null
aws sagemaker delete-model --model-name jio-traffic-predictor 2>/dev/null

# 4. CloudFront distribution (must disable first)
echo "[4/8] Disabling CloudFront distribution..."
# Disable via console — CLI requires ETag management (complex)
echo "   → Disable manually in CloudFront console, then delete"

# 5. S3 bucket (must empty first)
echo "[5/8] Emptying S3 bucket..."
aws s3 rm s3://${S3_BUCKET}/ --recursive 2>/dev/null
aws s3 rb s3://${S3_BUCKET}/ 2>/dev/null

# 6. DynamoDB tables
echo "[6/8] Deleting DynamoDB tables..."
aws dynamodb delete-table --table-name jio-capstone-sessions 2>/dev/null
aws dynamodb delete-table --table-name jio-capstone-content 2>/dev/null

# 7. Kinesis stream
aws kinesis delete-stream --stream-name jio-capstone-viewer-events 2>/dev/null

# 8. VPC resources (order matters!)
echo "[8/8] Cleaning VPC..."
source 01-networking/ids.env
# Delete security groups
aws ec2 delete-security-group --group-id ${ALB_SG} 2>/dev/null
aws ec2 delete-security-group --group-id ${EKS_SG} 2>/dev/null

  aws ec2 delete-subnet --subnet-id "$subnet" 2>/dev/null

# Delete VPC
aws ec2 delete-vpc --vpc-id ${VPC_ID} 2>/dev/null
